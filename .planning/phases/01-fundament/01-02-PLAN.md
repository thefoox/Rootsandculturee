       ```css
       /*
        * Motion guard: OPT-IN pattern per UI-SPEC Accessibility Contract.
        * Transitions/animations are OFF by default and only enabled when
        * user has no reduced-motion preference (prefers-reduced-motion: no-preference).
        * Components should use the `motion-safe:` Tailwind modifier for transition
        * classes, OR place their own transitions inside this media query.
        * This global block zeroes out any transitions that leak through.
        */
       *, *::before, *::after {
         animation-duration: 0ms;
         transition-duration: 0ms;
       }

       @media (prefers-reduced-motion: no-preference) {
         *, *::before, *::after {
           /* Re-enable transitions for users without a reduced-motion preference.
              Component-level durations (e.g. duration-100) now take effect. */
           animation-duration: revert;
           transition-duration: revert;
         }
       }
       ```

       IMPORTANT: The motion pattern uses `prefers-reduced-motion: no-preference` (opt-in) per UI-SPEC Accessibility Contract, NOT `prefers-reduced-motion: reduce` (opt-out). This means transitions are disabled by default and only restored when the user has expressed no preference for reduced motion. Components that use Tailwind transition classes (e.g. `transition-all duration-100`) will only animate when the user has no reduced-motion preference. This is the stricter WCAG-compliant approach.

    1. Create `src/components/ui/Button.tsx` -- a polymorphic button with 3 variants, all interaction states, and loading state:

       ```tsx
       'use client'

       import { forwardRef, type ButtonHTMLAttributes } from 'react'
       import { cn } from '@/lib/utils'
       import { Loader2 } from 'lucide-react'

       type ButtonVariant = 'primary' | 'secondary' | 'ghost'

       interface ButtonProps extends ButtonHTMLAttributes<HTMLButtonElement> {
         variant?: ButtonVariant
         loading?: boolean
       }

       const variantStyles: Record<ButtonVariant, string> = {
         primary: 'bg-ember text-cream hover:opacity-85 active:scale-[0.98]',
         secondary: 'border-2 border-forest text-forest hover:opacity-85 active:scale-[0.98]',
         ghost: 'text-forest hover:opacity-85 active:scale-[0.98]',
       }

       export const Button = forwardRef<HTMLButtonElement, ButtonProps>(
         ({ className, variant = 'primary', loading = false, disabled, children, ...props }, ref) => {
           const isDisabled = disabled || loading

           return (
             <button
               ref={ref}
               className={cn(
                 'inline-flex items-center justify-center gap-2 rounded-md px-4 py-2 font-body text-[15px] font-medium transition-all duration-100',
                 'min-h-[44px]',
                 variantStyles[variant],
                 isDisabled && 'opacity-40 cursor-not-allowed hover:opacity-40',
                 className
               )}
               disabled={isDisabled}
               aria-busy={loading || undefined}
               {...props}
             >
               {loading ? (
                 <>
                   <Loader2 className="h-4 w-4 animate-spin" aria-hidden="true" />
                   <span>Laster...</span>
                 </>
               ) : (
                 children
               )}
             </button>
           )
         }
       )

       Button.displayName = 'Button'
       ```

       Key implementation details:
       - `min-h-[44px]` ensures WCAG 2.5.5 touch target (per UI-SPEC Spacing Scale)
       - `variant='primary'` uses `bg-ember text-cream` (ember #B84D00 on cream, 4.98:1 contrast PASS AA)
       - Loading state replaces children with spinner + "Laster..." and sets `aria-busy="true"` (per UI-SPEC Interaction States)
       - Disabled state: `opacity-40 cursor-not-allowed` (per UI-SPEC)
       - Active state: `scale-[0.98]` with 100ms transition (per UI-SPEC)
       - Hover state: `opacity-85` on buttons (per UI-SPEC)
       - `transition-all duration-100` -- only animates when user has no reduced-motion preference (controlled by global opt-in guard in globals.css)

    2. Create `src/components/ui/Input.tsx` -- text input with integrated label and error state:

       ```tsx
       'use client'

       import { forwardRef, type InputHTMLAttributes } from 'react'
       import { cn } from '@/lib/utils'

       interface InputProps extends InputHTMLAttributes<HTMLInputElement> {
         label: string
         error?: string
       }

       export const Input = forwardRef<HTMLInputElement, InputProps>(
         ({ className, label, error, id, ...props }, ref) => {
           const inputId = id || label.toLowerCase().replace(/\s+/g, '-')
           const errorId = error ? `${inputId}-error` : undefined

           return (
             <div className="flex flex-col gap-1">
               <label
                 htmlFor={inputId}
                 className="text-[13px] font-normal tracking-wide text-forest"
               >
                 {label}
               </label>
               <input
                 ref={ref}
                 id={inputId}
                 className={cn(
                   'min-h-[44px] rounded-md border bg-card px-3 py-2 font-body text-[15px] text-forest placeholder:text-bark/60',
                   'transition-colors duration-100',
                   error
                     ? 'border-destructive'
                     : 'border-forest/20 focus:border-forest',
                   className
                 )}
                 aria-describedby={errorId}
                 aria-invalid={error ? true : undefined}
                 {...props}
               />
               {error && (
                 <p
                   id={errorId}
                   role="alert"
                   className="flex items-center gap-1 text-[13px] text-destructive"
                 >
                   {error}
                 </p>
               )}
             </div>
           )
         }
       )

       Input.displayName = 'Input'
       ```

       Key implementation details:
       - Every input has an associated `<label>` via `htmlFor`/`id` (never placeholder-only, per WCAG)
       - Error state: border changes to `border-destructive` (#C0392B), `aria-invalid="true"`, and `aria-describedby` links to inline error message (per WCAG-05)
       - Label uses 13px, tracking-wide (per UI-SPEC Typography: Label role)
       - Input background: `bg-card` (#F5F0E8 per UI-SPEC Color section)
       - `min-h-[44px]` for touch target (per WCAG 2.5.5)
