    Create `firestore.rules` in the project root with access control for all 6 collections defined in ARCHITECTURE.md. This file is deployed to Firebase via `firebase deploy --only firestore:rules` (user does this manually or via CI).

    The rules implement defense-in-depth (per ARCHITECTURE.md Pattern 5):
    - Helper functions: `isAuthenticated()`, `isAdmin()`, `isOwner(userId)`
    - Admin checks use Custom Claims (`request.auth.token.admin == true`), NOT Firestore document reads (per Pitfall 5 and Anti-Pattern 3)

    ```javascript
    rules_version = '2';
    service cloud.firestore {
      match /databases/{database}/documents {

        // --- Helper functions ---
        function isAuthenticated() {
          return request.auth != null;
        }
        function isAdmin() {
          return isAuthenticated() && request.auth.token.admin == true;
        }
        function isOwner(userId) {
          return isAuthenticated() && request.auth.uid == userId;
        }

        // --- Products: public read (published only), admin write ---
        match /products/{productId} {
          allow read: if resource.data.publishedAt != null;
          allow write: if isAdmin();
        }

        // --- Experiences: public read (published only), admin write ---
        match /experiences/{experienceId} {
          allow read: if resource.data.publishedAt != null;
          allow write: if isAdmin();

          // Dates subcollection: public read (availability), admin write
          // Atomic seat updates happen server-side via Admin SDK
          match /dates/{dateId} {
            allow read: if true;
            allow write: if isAdmin();
          }
        }

        // --- Orders: owner read, no client write (Admin SDK only) ---
        match /orders/{orderId} {
          allow read: if isOwner(resource.data.customerId) || isAdmin();
          allow write: if false;
        }

        // --- Bookings: owner read, no client write (Admin SDK only) ---
        match /bookings/{bookingId} {
          allow read: if isOwner(resource.data.customerId) || isAdmin();
          allow write: if false;
        }

        // --- Users: own profile CRUD, admin full access ---
        match /users/{userId} {
          allow read, update: if isOwner(userId);
          allow create: if isOwner(userId);
          allow read, write: if isAdmin();
        }

        // --- Blog: public read (published only), admin write ---
        match /blog/{articleId} {
          allow read: if resource.data.status == 'published';
          allow write: if isAdmin();
        }
      }
    }
    ```
